public class switchStatments {
	enum day {Mon, Tue, Wed, Thu, Fri, Sat, Sun};
	public static void main(String[] args) {
		myAccount NewAccount = new myAccount();
		myAccount NewAccount2 = new myAccount();
		myAccount NewAccount3 = new myAccount();
		NewAccount2.setday(day.Tue);
		NewAccount3.setday(day.Sun);
		getday(NewAccount.returnday());
		getday(NewAccount2.returnday());
		getday(NewAccount3.returnday());
	}
	 private static void getday(day returnday) {
		 switch (returnday) {
		  case Mon:
		    System.out.println("Monday");
		    break;
		  case Tue:
		    System.out.println("Tuesday");
		    break;
		  case Wed:
		    System.out.println("Wednesday");
		    break;
		  case Thu:
		    System.out.println("Thursday");
		    break;
		  case Fri:
		    System.out.println("Friday");
		    break;
		  case Sat:
		    System.out.println("Saturday");
		    break;
		  case Sun:
		    System.out.println("Sunday");
		    break;
		}
	}
	static class myAccount {
		day whatday = day.Mon;
		
		public void setday(day x) {
			this.whatday = x;
		}
			
		public day returnday() {
			return this.whatday;
		}
			
	}

}



OUTPUT:
 Monday
 Tuesday
 Sunday
